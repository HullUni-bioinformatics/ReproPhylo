# use "concat_must_have_all_of" once, and separate the loci with a space
# example: ,concat_must_have_all_of=coi 18S,
# use "concat_must_have_one_of" as many times as needed to create multiple loci groups,
# each needs to be represented by al least one locus.
# example: ,concat_must_have_one_of=18S 28S,concat_must_have_one_of=alg11 ndh5,
set name=combined,loci=all,concat_meta=concatenation_dict,concat_must_have_all_of=coi,concat_must_have_one_of=18S 28S
